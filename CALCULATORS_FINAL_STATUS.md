// 真实Free Space Path Loss公式 (ITU-R P.525-4)
FSPL (dB) = 20 × log₁₀(distance) + 20 × log₁₀(frequency_MHz) - 27.55

// 链路预算分析
RSSI (dBm) = TX_Power - Total_Path_Loss
